Code Pattern

CADD 파이프라인에서 자주 쓰는 패턴 — smina 도킹 결과 SDF를 파싱하고 fused ring system 기준으로 히트를 분류하는 예시입니다.

hit_filter.py
from pathlib import Path
from rdkit import Chem
from rdkit.Chem import rdMolDescriptors
import pandas as pd

# Union-Find로 fused ring system 크기 탐지
def get_max_fused_ring_size(mol: Chem.Mol) -> int:
    ri = mol.GetRingInfo()
    atom_rings = ri.AtomRings()
    if not atom_rings:
        return 0

    parent = {a: a for ring in atom_rings for a in ring}

    def find(x):
        while parent[x] != x:
            parent[x] = parent[parent[x]]
            x = parent[x]
        return x

    for r1 in atom_rings:
        for r2 in atom_rings:
            if set(r1) & set(r2):
                parent[find(r1[0])] = find(r2[0])

    groups: dict = {}
    for ring in atom_rings:
        root = find(ring[0])
        groups.setdefault(root, set()).add(id(ring))

    return max(len(v) for v in groups.values())


def filter_hits(sdf_path: Path, threshold: int = 3) -> pd.DataFrame:
    """SDF에서 fused ring ≥ threshold인 히트만 추출."""
    records = []
    supplier = Chem.SDMolSupplier(str(sdf_path), removeHs=False)

    for mol in supplier:
        if mol is None:
            continue
        score = float(mol.GetProp("minimizedAffinity"))
        fused  = get_max_fused_ring_size(mol)
        if fused >= threshold:
            records.append({
                "name":  mol.GetProp("_Name"),
                "score": score,
                "fused_rings": fused,
                "smiles": Chem.MolToSmiles(mol),
            })

    return pd.DataFrame(records).sort_values("score")
